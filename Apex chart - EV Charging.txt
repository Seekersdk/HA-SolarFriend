    yaxis_id: price

    unit: kr
    data_generator: |
      const raw = [
        ...(entity.attributes.raw_today || []),
        ...(entity.attributes.raw_tomorrow || [])
      ];
      const seen = new Set();
      return raw
        .filter(h => {
          const t = new Date(h.hour).setMinutes(0, 0, 0);
          if (seen.has(t)) return false;
          seen.add(t);
          return true;
        })
